Configuración

Conexión a Supabase

🔫 Conectar Supabase

Ingresá la URL y la clave anon de tu proyecto. Los datos se guardan localmente en el navegador.

📋 SQL para crear las tablas

Ejecutá esto en el SQL Editor de tu proyecto Supabase:

CREATE TABLE ventas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha date NOT NULL, presupuesto text, factura text, cliente text,
  tipo text CHECK (tipo IN ('Servicio','Producto')),
  descripcion text, monto numeric NOT NULL, iva boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE compras (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha date NOT NULL, presupuesto text, factura text, proveedor text,
  tipo text CHECK (tipo IN ('Servicio','Producto')),
  descripcion text, monto numeric NOT NULL, iva boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE presupuestos (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha date NOT NULL, numero text, cliente text, descripcion text,
  monto numeric, estado text DEFAULT 'Pendiente'
    CHECK (estado IN ('Pendiente','Aprobado','Rechazado')),
  notas text, created_at timestamptz DEFAULT now()
);
CREATE TABLE gastos_fijos (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  concepto text NOT NULL, categoria text, monto_mensual numeric NOT NULL,
  frecuencia text DEFAULT 'Mensual', activo boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);
-- RLS
ALTER TABLE ventas ENABLE ROW LEVEL SECURITY;
ALTER TABLE compras ENABLE ROW LEVEL SECURITY;
ALTER TABLE presupuestos ENABLE ROW LEVEL SECURITY;
ALTER TABLE gastos_fijos ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all" ON ventas FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "all" ON compras FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "all" ON presupuestos FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "all" ON gastos_fijos FOR ALL USING (true) WITH CHECK (true);

-- COBROS PENDIENTES
CREATE TABLE cobros_pendientes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha date NOT NULL,
  presupuesto_id uuid,
  presupuesto_num text,
  factura text,
  cliente text NOT NULL,
  descripcion text,
  monto numeric NOT NULL,
  estado text DEFAULT 'Pendiente'
    CHECK (estado IN ('Pendiente','Cobrado','Vencido')),
  vencimiento date,
  notas text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE cobros_pendientes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all" ON cobros_pendientes FOR ALL USING (true) WITH CHECK (true);
